def result_analysis(y: np.ndarray, y_pred: np.ndarray): #Fonction Analyse de Prcision, renvoi recall, prcision et F-Score 
    Recall = recall_score(y,y_pred) #true positif/(true positif+false negatif)
    Precision = precision_score(y,y_pred) #=true positif/(true positif+false positive)
    Fscore = 2*(Precision*Recall)/(Precision+Recall) #Classique Classifier score
    return(Recall, Precision, Fscore)

def Confusion(y: np.ndarray, y_pred: np.ndarray): #Renvoie la matrice de Confusion pour notre prdiction ( utiliser que pour avoir l'image de la matrice de confusion, donc sur le meilleur classifier)
    Conf = confusion_matrix(y, y_pred)
    Confshow = skplt.metrics.plot_confusion_matrix(y, y_pred, normalize=True)
